2. The Logic (script.js) Create this file in the same folder. This script tracks the total and adds a list item every time you click. JavaScript let total = 0; const cartElement = document.getElementById('shopping-cart'); const totalSpan = document.getElementById('total-span'); const snackButtons = document.querySelectorAll('.snack-btn'); const clearBtn = document.getElementById('clear-cart'); // Function to handle clicking a snack snackButtons.forEach(button => { button.addEventListener('click', () => { const name = button.getAttribute('data-name'); const price = parseFloat(button.getAttribute('data-price')); // 1. Update Total total += price; totalSpan.innerText = total.toFixed(2); // 2. Add to List const listItem = document.createElement('li'); listItem.className = "bg-gray-700 py-2 px-4 rounded inline-block mx-1 mb-1"; listItem.textContent = `${name} - $${price.toFixed(2)}`; cartElement.appendChild(listItem); }); }); // Clear button logic clearBtn.addEventListener('click', () => { total = 0; totalSpan.innerText = "0.00"; cartElement.innerHTML = ""; });
